ndarray

  • NumPy’s array class is called ndarray
  • all elements must be the same type
import numpy as np
np.__version__
#  '1.19.0'
# create an array from a regular Python list 
myarray = np.array([[ 1., 2., 3.],  
                    [ 4., 5., 6.]])
print(myarray)
#  [[1. 2. 3.]
#   [4. 5. 6.]]

attributes

myarray.ndim
#  2
myarray.shape
#  (2, 3)
myarray.dtype
#  dtype('float64')
myarray.size
#  6
myarray.itemsize
#  8
myarray.data
#  <memory at 0x7f87f0da3d40>

indexing

# first row, first column
myarray[0,0]
#  1.0
myarray[1,1]
#  5.0
myarray[:2, 1:3]
#  array([[2., 3.],
#         [5., 6.]])

methods

myarray.reshape((3,2))
#  array([[1., 2.],
#         [3., 4.],
#         [5., 6.]])
myarray.ravel()

# myarray.copy() # deep copy

Note the difference between np.ndarray and np.array() . The former is an actual data type, while the latter is a function to make arrays from other data structures
