echo '<mydocument has="an attribute">
  <and>
    <many>elements</many>
    <many>more elements</many>
  </and>
  <plus a="complex">
    element as well
  </plus>
</mydocument>' > temp.xml
from pprint import pprint
import xmltodict
with open('temp.xml') as fh:
    doc = xmltodict.parse(fh.read())
pprint(doc)
doc['mydocument']['@has']
doc['mydocument']['and']['many']
doc['mydocument']['plus']['@a']
doc['mydocument']['plus']['#text']
