Series

a one-dimensional labeled array capable of holding any data type

create

pd.Series(data, index, dtype, copy)

ser = pd.Series() # empty series

from scalar

ser = pd.Series(0) 
ser = pd.Series(0, ['a', 'b', 'c'])

from list

ser = pd.Series([1, 2, 3])
ser = pd.Series([1, 2, 3], ['a', 'b', 'c'])

from dict

ser = pd.Series({'a': 1, 'b': 2, 'c': 3})

from ndarray

ser = pd.Series(np.array([1, 2, 3]), ['a', 'b', 'c'])

by location

ser[0] # first (returns scalar)
ser[-1] # last (returns scalar)
ser[[0]] # first
ser[[-1]] # last
ser[[0,2]] # first and third
ser[1:] # everything but the first

ser[-1:] # last

ser['c'] # last (returns scalar)

ser[['c']] # last

ser['b':] # everything but the first
ser[:'b'] # everything but the last
ser[:'a'] # first
ser['c':] # last

DataFrame

a two-dimensional labeled data structure with columns of potentially different types

create

pd.DataFrame(data, index, columns, dtype, copy)

# create empty df
df = pd.DataFrame()

from dict

# create from dict of lists - each key is a column
df = pd.DataFrame({'v1': ['x', 'y', 'z'],
                  'v2': [1, 2, 3]},
                 index=['a', 'b', 'c'])

# create from dict of series - each key is a column
df = pd.DataFrame({'v1': pd.Series(['x', 'y', 'z'], index=['a', 'b', 'c']),
                  'v2': pd.Series([1, 2, 3], index=['a', 'b', 'c'])})

from list

df = pd.DataFrame([1, 2, 3])
df = pd.DataFrame([1, 2, 3], columns=['v1'])

# create from list of lists - each sublist is a row
df = pd.DataFrame([['x', 1], ['y', 2], ['z', 3]],
                 index=['a', 'b', 'c'],
                 columns=['v1', 'v2'])

# create from list of dicts - each sublist is a row
df = pd.DataFrame([{'v1': 'x', 'v2': 1},
                  {'v1': 'y', 'v2': 2},
                  {'v1': 'z', 'v2': 3}],
                 index=['a', 'b', 'c'])

attributes

df.shape # (nrows, ncols) tuple
df.head # first 5 rows
df.tail # last 5 rows

indexing and slicing

subset rows by index

df.loc['a'] # first row - returns a series
df.loc["c"] # last row - returns a series
df.loc[["a"]] # first row
df.loc[["c"]] # last row

subset rows by position

df.iloc[0] # first row - returns a series
df.iloc[-1] # last row - returns a series
df.iloc[[0]] # first row
df.iloc[[-1]] # last row

subset rows by boolean

df[df['v1'] == "x"]

Or a more generalisable form,

df.loc[df['v1'].isin(['x', 'y'])]

subset columns by name

df['v1'] # returns a series
df[['v1']]
df[['v1', 'v2']]

Equivalently,

df.loc[:, 'v1']  # returns a series
df.loc[:, ['v1']]
df.loc[:, ['v1', 'v2']]

subset columns by position

df.iloc[:, 0]  # returns a series
df.iloc[:, [0]]
df.iloc[:, [0, 1]]

subset columns by boolean

df.loc[:, [True, False]]
df.loc[:, [True, True]]

df.sort_values(by='col1')

df.sort_values(by='col1', ascending=False, na_position='first')
